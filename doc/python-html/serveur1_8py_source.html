22  def getDataLines(self, columns, table=None):
23  """
24  Effectue une requête dans la base de données et renvoie un itérable
25  contenant les données demandées
26  @param columns identifiants de colonnes
27  @param table identifiant de tableau (inutile ici)
28  """
29  result=[]
30  for perle in self.list:
31  l=[]
32  for c in columns:
33  if len(c) == 2 and isinstance(c[0], types.FunctionType):
34  # on doit appliquer la fonction c[0] aux arguments
35  # pointés par c[1]
36  args=[]
37  for argName in c[1]:
38  if isinstance(argName, str):
39  val=getattr(perle, argName, None)
40  args.append(val)
41  else:
42  args.append(argName)
43  l.append(c[0](*args))
44  else:
45  val=getattr(perle, c, None)
46  l.append(val)
47  result.append(l)
