12 class TableFormat(object):
13  """
14  Une classe pour préciser un format de tables à afficher en HTML,
15  lié à une requête de base de données.
16  On peut faire des classes dérivées de celle-ci pour fabriquer des
17  tableaux avec des propriétés particulières.
18  """
19  def __init__(self, tname, tclasse, columns, formats, names, classes, header=True):
20  """
21  Constructeur
22  @param tname nom de la table dans la BDD
23  @param tclasse classe CSS pour le tableau
24  @param columns un tuple de colonnes de bases de données
25  @param formats un tuple de chaînes de format pour les données
26  @param names noms des contenus des colonnes pour faire des titres
27  @param classes classes CSS pour les colonnes
28  @param header vrai (par défaut) si on veut une ligne de titres
29  """
30  ## nom de la table dans la BDD
31  self.tname=tname
32  ## classe CSS à appliquer au tableau
33  self.tclasse=tclasse
34  ## noms de colonnes de la table dans la BDD
35  self.columns=columns
36  ## chaînes de formats applicables pour afficher ces colonnes
37  self.formats=formats
38  ## titres compréhensibles par des humains pour les colonnes
39  self.names=names
40  ## classes CSS à appliquer aux données de ces colonnes
41  self.classes=classes
42  ## booléen : présence/absence de titres pour l'affichage des colonnes
43  self.header=header
44  return
45 
46  def entete(self):
47  """
48  Fabrique une ligne de titres
49  @return une ligne de titres de colonnes avec les bonnes classes CSS
50  """
51  result= u"<tr>"
52  for cn in zip(self.classes, self.names):
53  result += u"<th class='%s'>%s</th>" % cn
54  result+= u"</tr>"
55  return result

57  def tbody(self,c):
58  """
59  Fabrique le corps du tableau
60  @param c un curseur de BDD déjà initialisé
61  @return une suite de lignes pour un tableau HTML
62  """
63  result=""
64  request="SELECT " + ", ".join(self.columns) + " FROM "+ self.tname
65  for ligne in c.execute(request):
66  result+= u"<tr>"
67  for li,fo,cl in zip(ligne, self.formats, self.classes):
68  fstring=u"<td class='%%s'>%s</td>" % fo
69  result += fstring % (cl,li)
70  result+= u"</tr>\n"
71  return result

74  """
75  Fabrique le tableau HTML
76  @param c un curseur de BDD déjà initialisé
77  @return tableau HTML
78  """
79  result = u"<table>\n"
80  if self.header:
81  result += self.entete()
82  result += self.tbody(c)
83  result += "</table>\n"
