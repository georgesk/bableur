  2 /**
  3  * Fonction pour traduire un texte court en appelant le
  4  * service http://mymemory.translated.net
  5  * @param listenum une liste de numéros de textes à traduire
  6  * @param l liste de langues correspondant à ces textes
  7  * @param endl null ou la langue de départ si on veut une
  8  *  traduction circulaire
  9  **/
 10 function traduire(listenum, l, endl){
 11   var o = $('#area'+ listenum[0]);
 12   var t = $('#area'+ listenum[1]);
 13   var n = listenum[1];              // numéro du texte destination
 14   var langues = l[0]+'|'+l[1];
 15 
 16   if (l.length == 0) return;        // traduction circulaire finie
 17   if (l.length == 1) {              // dernière langue
 18     if (endl === undefined) return; // traduction non circulaire
 19     else {                          // traduction circulaire demandée
 20       t = $('#area0');
 21       n = 0;
 22       langues = l[0]+'|'+endl;
 23     }
 24   }
 25 
 26   var nextList = listenum.slice(1);
 27   var nextL       = l.slice(1);
 28   var oldText = t.val();
 29   t.val("");
 30   setWaiting(t, true);                       // début de l'attente
 31   $.getJSON(
 32     'http://mymemory.translated.net/api/get',// service de traduction
 33     { q: o.val(),                            // paramètres
 34       langpair : langues,
 35       de: "georges.khaznadar@free.fr"})      // dont un e-mail valide
 36     .done(function(data){                    // rappel en cas de réussite
 37       var decoded =                          // décode les entités HTML
 38       $("<div/>").html(data.responseData.translatedText).text();
 39       t.val(decoded);                        //écrit la traduction
 40       stackOldText(oldText, decoded, n);     // range l'ancien texte
 41       setWaiting(t, false);                  // fin de l'attente 
 42       traduire(nextList, nextL, endl);       // appel récursif
 43     })
 44     .fail(function(){                        // rappel en cas d'échec
 45       t.val("** Échec de la traduction **"); // message d'erreur
 46       setWaiting(t, false);                  // fin de l'attente
 47       return;
 48     });
 49   return;
 50 }

 52 /**
 53  * empile l'ancien texte d'une traduction s'il est non vide, sous
 54  * le champ texte modifié.
 55  * @param oldtext l'ancien texte
 56  * @param newtext le nouveau texte
 57  * @param n le numéro du champ
 58  **/
 59 function stackOldText(oldtext, newtext, n){
 60   if (oldtext.length > 0 && oldtext != newtext){
 61     var stack = $('#pile'+n);
 62     var li    = $("<li>", {class: 'lienDrole'})
 63       .text(oldtext)
 64       .click(function(){conserver(oldtext, newtext)});
 65     stack.prepend(li);
 66   }
 67 }

 69 /**
 70  * Conserve, si on veut, un ancien texte et son remplaçant.
 71  * @param ancienTexte l'ancien texte
 72  * @param nouveauTexte le nouveau texte
 73  **/
 74 function conserver(ancienTexte, nouveauTexte){
 75   var ok    = confirm("Voulez-vous conserver ce texte ?\n"+ancienTexte+"\n   ---->\n"+nouveauTexte);
 76   if (ok) {
 77     var archiveCSV='"'+ancienTexte+
 78       '"; "'+nouveauTexte+
 79       '"; "'+$("#circulaire").val()
 80       +'"';
 81     $("#archive").append($("<li>").text(archiveCSV));
 82   }
 83 }

 85 /**
 86  * Fonction pour faire apparaître un élément comme "en attente".
 87  * @param el l'objet jQuery à modifier
 88  * @param state booléen ; vrai => "en attente", faux => "normal"
 89  **/
 90 function setWaiting(el,state){
 91   // l'usage de setTimeout permet de forcer la modification
 92   // immédiate de l'interface utilisateur ; sans ça les
 93   // changements de style n'ont pas lieu à temps.
 94   if(state){
 95     setTimeout(function(){
 96       el.addClass("busy");
 97     },0);
 98   } else {
 99     setTimeout(function(){
100       el.removeClass("busy");
101     },0);
102   }
103 }

154 /**
155  * Initialisation de la chaîne de traductions.
156  * @param l une liste de langues ; si cette liste est indéfinie, les
157  *        langues définies par les étiquettes mobiles sont prises en compte
158  * @param target un div où construire la chaîne de langues la première fois
159  *        avant que la page n'ait été affichée
160  **/
161 function chaineTrad_init(l, target){
162   if (l == undefined) {
163     var chaineTrad=$("#langChain").find(".drapeau");
164     var l=Array();
165     chaineTrad.each(function(i,val){
166       l.push($(val).text());
167     });
168     if (l.length < 2){
169       alert("Il faut deux langues au moins pour faire une traduction.\nAjoutez des langues dans la colonne de gauche.")
170     }
171     var trad = $("#translationBox"); // on récupère la boîte de traductions
172     trad.empty();                    // et on la vide
173   } else {                           // ou alors on crée la boîte de traductions
174     var trad = $("<div>", {id: "translationBox"})
175     target.append(trad);
176   }
177   trad.append(roundTranslateButton(l));
178   trad.append($("<br>"));
179   for (var i=0; i < l.length; i++){
180     trad.append(fsTraduction(l,i));
181   }
182   $("#area0").val("Entrez ici un texte à traduire");
183 }

186 /**
187  * Crée un objet jQuery comprenant un bouton pour déclencher
188  * une série circulaire de traductions.
189  * l une liste de langues
190  * @return l'objet jQuery prêt à servir
191  **/
192 function roundTranslateButton(l){
193   var listeCirculaire=l.slice(0); // recopie
194   listeCirculaire.push(l[0]);     // circularisation
195   var text="Traduction circulaire : "+listeCirculaire.join(" -> ");
196   var listenum=Array();
197   for(var i=0; i<l.length; i++) listenum.push(i);
198   var script="traduire("+JSON.stringify(listenum)+", "+JSON.stringify(l)+", '"+l[0]+"')";
199   var button=$(
200     "<input>",
201     { type: "button",
202       onclick: script,
203       value: text,
204       id : "circulaire"
205     }
206   );
207   return button;
208 }

260 /**
261  * Crée un objet jQuery comprenant un bouton et un champ de texte
262  * le champ de texte possède un identifiant numéroté : "area" +n.
263  * En plus, une boîte avec une liste numérotée est ajouté pour
264  * contenir les anciennes versions du texte.
265  * @param l une liste de langues
266  * @param n l'index sur la première langue à considérer dans la liste
267  * @return l'objet jQuery prêt à servir
268  **/
269 function fsTraduction(l,n){
270   var fieldset;
271   var legendText = "Langue : "+l[n]+" ";
272   if (n == 0){
273     legendText = "Langue de départ : "+l[n]+" ";
274   }
275   fieldset=$("<fieldset>",{class: "translator"});
276   fieldset.append($("<legend>").text(legendText)
277                   .append($("<img>",{src:"flags/"+l[n]+".png",
278                                    height:"10px",
279                                      alt:"drapeau "+l[n]}))
280                  );
281   fieldset.append($("<input>", 
282 		    {type: "button", 
283 		     value: evoqueTraduire(l,n),
284 		     onclick: invoqueTraduire(l,n)
285 		    })
286 		 );
287   fieldset.append($("<br>"));
288   fieldset.append($("<textarea>",
289 		    {cols: "40",
290 		     rows: "4",
291 		     id: "area"+n})
292 		 );
293   fieldset.append($("<ol>",{id : "pile"+n, class: "ancienneTraduction"}));
294   return fieldset;
295 }

297 /**
298  * crée un script pour traduire d'une langue vers une autre
299  * d'après une liste et un index. Si l'index pointe à la fin
300  * de la liste, on traduit de la dernière vers la première langue.
301  * @param l une liste de langues
302  * @param n index sur cette liste
303  * @return le texte du script qui fait la traduction
304  **/
305 function invoqueTraduire(l,n){
306   var from = n
307   var to   = (n+1) % l.length;
308   var callback="traduire("+
309     JSON.stringify([from, to])+
310     ", "+
311     JSON.stringify([l[from],l[to]])+
312     ");"
313   return callback;
314 }

316 /**
317  * crée un texte pour évoquer la traduction d'une langue vers une autre,
318  * d'après une liste de langues et un index. Si l'index pointe à la fin
319  * de la liste, on traduit de la dernière vers la première langue.
320  * @param l une liste de langues
321  * @param n index sur cette liste
322  * @return le texte évoquant la traduction
323  **/
324 function evoqueTraduire(l,n){
325   var from = n
326   var to   = (n+1) % l.length;
327   return "Traduire : "+[l[from],l[to]].join(" -> ");
328 }
